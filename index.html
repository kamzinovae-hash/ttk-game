# ========================================= # 🤖 TELEGRAM BOT SYSTEM (TTK PROJECT) # AIogram 3.x # ========================================= from aiogram import Bot, Dispatcher, types from aiogram.types import ReplyKeyboardMarkup, KeyboardButton from aiogram.filters import Command import asyncio import json # ========================================= # 🔑 ТОКЕН БОТА # ========================================= TOKEN = "ВСТАВЬ_ТОКЕН_СЮДА" bot = Bot(token=TOKEN) dp = Dispatcher() # ========================================= # 🧠 ПАМЯТЬ ПОЛЬЗОВАТЕЛЕЙ (баллы) # ========================================= users = {} def get_user(user_id: int): if user_id not in users: users[user_id] = { "score": 0, "balance": 0, "bonuses": [] } return users[user_id] # ========================================= # 📋 ГЛАВНОЕ МЕНЮ # ========================================= main_menu = ReplyKeyboardMarkup( keyboard=[ [KeyboardButton(text="📄 Профиль")], [KeyboardButton(text="🧾 Заявки")], [KeyboardButton(text="🎁 Бонусы")], [KeyboardButton(text="🎮 Играть")] ], resize_keyboard=True ) # ========================================= # 🚀 START # ========================================= @dp.message(Command("start")) async def start(message: types.Message): get_user(message.from_user.id) await message.answer( "🎮 TTK SYSTEM\nВыберите раздел:", reply_markup=main_menu ) # ========================================= # 📄 ПРОФИЛЬ # ========================================= @dp.message(lambda m: m.text == "📄 Профиль") async def profile(message: types.Message): user = get_user(message.from_user.id) await message.answer( f"👤 Профиль\n\n" f"⭐ Баллы: {user['score']}\n" f"💰 Баланс: {user['balance']}\n" f"🎁 Бонусов: {len(user['bonuses'])}" ) # ========================================= # 🧾 ЗАЯВКИ # ========================================= @dp.message(lambda m: m.text == "🧾 Заявки") async def tickets(message: types.Message): await message.answer( "🧾 Поддержка:\n" "• Ошибки\n" "• Вопросы\n" "• Обращения" ) # ========================================= # 🎁 БОНУСЫ # ========================================= @dp.message(lambda m: m.text == "🎁 Бонусы") async def bonuses(message: types.Message): user = get_user(message.from_user.id) await message.answer( "🎁 МАГАЗИН БОНУСОВ\n\n" "💸 Скидка 5% — 50💰\n" "📶 +ГБ интернета — 100💰\n" "📞 Минуты — 80💰\n" "🎬 Kino 7 дней — 120💰\n" "⚡ Приоритет — 70💰\n\n" f"💰 Ваш баланс: {user['balance']}" ) # ========================================= # 🎮 ИГРА (WEBAPP) # ========================================= @dp.message(lambda m: m.text == "🎮 Играть") async def game(message: types.Message): kb = types.InlineKeyboardMarkup(inline_keyboard=[ [types.InlineKeyboardButton( text="▶ Играть", web_app=types.WebAppInfo( url="https://kamzinovae-hash.github.io/ttk-game/" ) )] ]) await message.answer("Запуск игры:", reply_markup=kb) # ========================================= # 📥 ПРИЁМ ДАННЫХ ИЗ ИГРЫ # ========================================= @dp.message() async def game_data(message: types.Message): try: data = json.loads(message.text) user = get_user(message.from_user.id) score = data.get("score", 0) balance = data.get("balance", 0) # 💰 обновляем данные user["score"] = score user["balance"] += balance await message.answer( f"🎮 Результат принят!\n" f"⭐ +{score} очков\n" f"💰 +{balance} баллов" ) except: pass # ========================================= # 🔁 ЗАПУСК # ========================================= async def main(): await dp.start_polling(bot) if __name__ == "__main__": asyncio.run(main())
